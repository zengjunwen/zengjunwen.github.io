class Solution {
    int[] memo;
    public int fib(int N) {
        if (N == 0||N==1) {
            return N;
        }
        memo = new int[N + 1];
        Arrays.fill(memo, Integer.MAX_VALUE);
        memo[0] = 0;
        memo[1] = 1;
        return traversal(N);
    }

    private int traversal(int N) {
        if (N == 0||N==1) {
            return N;
        }
        int a,b;
        a = memo[N-1] == Integer.MAX_VALUE?traversal(N-1):memo[N-1];
        b = memo[N-2] == Integer.MAX_VALUE?traversal(N-2):memo[N-2];
        memo[N] = a + b;
        return memo[N];
    }
}

class Solution {
    public int fib(int N) {
        if(N < 2) {
            return N;
        }

        int[] dp = new int[N + 1];
        dp[0] = 0;
        dp[1] = 1;
        for(int i = 2; i <= N; i++) {
            dp[i] = dp[i-1] + dp[i-2];
        }
        
        return dp[N];
    }
}
